CMAKE_MINIMUM_REQUIRED(VERSION 3.10)
PROJECT(dynet)
SET(CMAKE_CXX_STANDARD 11)
ADD_LIBRARY(calc SHARED ${PROJECT_SOURCE_DIR}/calc.cpp)
ADD_EXECUTABLE(${PROJECT_NAME} ${PROJECT_SOURCE_DIR}/main.cc)
TARGET_LINK_LIBRARIES(${PROJECT_NAME} calc)

IF(ENABLE_ADD)
    target_compile_definitions(calc PRIVATE EXEC_ADD)
    target_compile_definitions(${PROJECT_NAME}  PRIVATE EXEC_ADD)
ENDIF()
